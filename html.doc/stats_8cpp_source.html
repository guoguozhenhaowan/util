130 void Stats::summarize(bool forced){
131  if(this->summarized && !forced){
132  return;
133  }
134 
135  // cycle and total bases
136  bool getMinReadLen = false;
137  int c = 0;
138  for(c = 0; c < this->bufLen; ++c){
139  this->bases += this->cycleTotalBase[c];
140  if(!getMinReadLen && c > 1 && this->cycleTotalBase[c] < this->cycleTotalBase[c-1]){
141  this->minReadLen = c;
142  getMinReadLen = true;
143  }
144  if(this->cycleTotalBase[c] == 0){
145  break;
146  }
147  }
148  this->cycles = c;
149  this->maxReadLen = c;
150 
151  // Q20, Q30, base content
152  for(int i = 0; i < 8; ++i){
153  for(int j = 0; j < this->cycles; ++j){
154  this->q20Bases[i] += this->cycleQ20Bases[i][j];
155  this->q30Bases[i] += this->cycleQ30Bases[i][j];
156  this->baseContents[i] += this->cycleBaseContents[i][j];
157  }
158  this->q20Total += this->q20Bases[i];
159  this->q30Total += this->q30Bases[i];
160  }
161 
162  // quality curve for mean qual
163  double* meanQualCurve = new double[this->cycles];
164  std::memset(meanQualCurve, 0, sizeof(double) * this->cycles);
165  for(int i = 0; i < this->cycles; ++i){
166  meanQualCurve[i] = (double)this->cycleTotalQual[i] / (double)this->cycleTotalBase[i];
167  }
168  this->qualityCurves["mean"] = meanQualCurve;
169 
170  // quality curves and base contents curves for different nucleotides
171  char nucleotides[5] = {'A', 'T', 'C', 'G', 'N'};
172  for(int i = 0; i < 5; ++i){
173  char b = nucleotides[i] & 0x07;
174  double* qualCurve = new double[this->cycles];
175  std::memset(qualCurve, 0, sizeof(double) * this->cycles);
176  double* contentCurve = new double[this->cycles];
177  std::memset(contentCurve, 0, sizeof(double) * this->cycles);
178  for(int j = 0; j < this->cycles; ++j){
179  if(this->cycleBaseContents[b][j] == 0){
180  qualCurve[j] = meanQualCurve[j];
181  }else{
182  qualCurve[j] = (double)this->cycleBaseQual[b][j] / (double)this->cycleBaseContents[b][j];
183  }
184  contentCurve[j] = (double)this->cycleBaseContents[b][j] / (double)this->cycleTotalBase[j];
185  }
186  this->qualityCurves[std::string(1, nucleotides[i])] = qualCurve;
187  this->contentCurves[std::string(1, nucleotides[i])] = contentCurve;
188  }
189 
190  // GC content curve
191  double* gcContentCurve = new double[this->cycles];
192  std::memset(gcContentCurve, 0, sizeof(double) * this->cycles);
193  char gIndex = 'G' & 0x07;
194  char cIndex = 'C' & 0x07;
195  for(int i = 0; i < this->cycles; ++i){
196  gcContentCurve[i] = (double)(this->cycleBaseContents[gIndex][i] + this->cycleBaseContents[cIndex][i])/ (double)this->cycleTotalBase[i];
197  }
198  this->contentCurves["GC"] = gcContentCurve;
199 
200  if(this->kmerLen){
201  // K-mer statistics
202  this->kmerMin = this->kmer[0];
203  this->kmerMax = this->kmer[0];
204  for(int i = 1; i < this->kmerBufLen; ++i){
205  this->kmerMin = std::min(this->kmer[i], this->kmerMin);
206  this->kmerMax = std::max(this->kmer[i], this->kmerMax);
207  }
208  }
209 
210  this->summarized = true;
211 }

220 void Stats::statRead(Read* r){
221  int len = r->length();
222  this->lengthSum += len;
223  if(this->bufLen < len){
224  this->extendBuffer(std::max(len + 100, (int)(len * 1.5)));
225  }
226  const std::string seqStr = r->seq.seqStr;
227  const std::string qualStr = r->quality;
228  int kmerVal = -1;
229  for(int i = 0; i < len; ++i){
230  char base = seqStr[i];
231  char qual = qualStr[i];
232  char bIndex = base & 0x07;
233  const char q20 = '5';
234  const char q30 = '?';
235  this->minQual = std::min(qual - 33, this->minQual);
236  this->maxQual = std::max(qual - 33, this->maxQual);
237  if(qual > q30){
238  ++this->cycleQ20Bases[bIndex][i];
239  ++this->cycleQ30Bases[bIndex][i];
240  }else if(qual > q20){
241  ++(this->cycleQ20Bases[bIndex][i]);
242  }
243 
244  ++this->cycleBaseContents[bIndex][i];
245  this->cycleBaseQual[bIndex][i] += (qual - 33);
246  ++this->cycleTotalBase[i];
247  this->cycleTotalQual[i] += (qual - 33);
248 
249  if(this->kmerLen){
250  if(i < this->kmerLen - 1){
251  continue;
252  }
253  kmerVal = Evaluator::seq2int(seqStr, i - this->kmerLen + 1, this->kmerLen, kmerVal);
254  if(kmerVal >= 0){
255  ++this->kmer[kmerVal];
256  }
257  }
258  }
259 
260  if(this->overRepSampleFreq){
261  if(this->reads % this->overRepSampleFreq == 0){
262  std::set<int> steps = {10, 20, 40, 100, std::min(150, this->evaluatedSeqLen - 2)};
263  for(auto& step: steps){
264  for(int j = 0; j < len - step; ++j){
265  std::string seq = seqStr.substr(j, step);
266  if(this->overRepSeq.count(seq) > 0){
267  ++this->overRepSeq[seq];
268  for(int p = j; p < seq.length() + j && p < this->evaluatedSeqLen; ++p){
269  ++this->overRepSeqDist[seq][p];
270  }
271  j += step;
272  }
273  }
274  }
275  }
276  }
277  ++this->reads;
278 }
